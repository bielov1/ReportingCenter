                        ];
        
                        function getTagCount(tweets) {
                            const tagCount = {};
        
                            tweets.forEach(tweet => {
                                tweet.tags.forEach(tag => {
                                    if (tagCount[tag]) {
                                        tagCount[tag]++;
                                    } else {
                                        tagCount[tag] = 1;
                                    }
                                });
                            });
        
                            return tagCount;
                        }
        
                        const result = getTagCount(tweets);
                        console.log(result);

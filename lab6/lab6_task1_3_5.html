JavaScript код

                        const inputText = document.getElementById("inputText3");
                        const toggleButton = document.getElementById("toggleBtn");
                        let hiddenText = "";
              
                        function toggleText() {
                            const text = inputText.value.trim();
              
                            if (text === "") {
                                alert("Будь ласка, введіть текст, поле порожнє");
                                return;
                            }
              
                            if (hiddenText === "") {
                                hiddenText = text;
                                inputText.value = "*".repeat(text.length);
                                toggleButton.textContent = "Розкрити";
                            } else {
                                inputText.value = hiddenText;
                                hiddenText = "";
                                toggleButton.textContent = "Приховати";
                            }
                        }
              
                        toggleButton.addEventListener("click", toggleText);
